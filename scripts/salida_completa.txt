export class HeaderComponent implements OnInit {
  isScrolled = false;

  constructor(private navigationService: NavigationService) {}

  ngOnInit(): void {}

  @HostListener('window:scroll', [])
  onWindowScroll(): void {
    this.isScrolled = window.scrollY > 100;
  }

  navigateToSection(section: string, event: Event): void {
    event.preventDefault();
    this.navigationService.navigateToSection(section);
    
    if (section !== 'inicio') {
      window.scrollTo({
        top: 0,
        behavior: 'smooth'
      });
    }
  }

  toggleMobileMenu(): void {

  }

export class AdminDashboardComponent implements OnInit, OnDestroy {
  Math = Math;
  totalProducts = 0;
  activeProducts = 0;
  totalCategories = 0;
  productsWithDiscount = 0;
  mostPopularCategory = '';
  whatsappNumber = '';
  
  recentActivities = [
    {
      type: 'system',
      icon: 'bi bi-check-circle-fill',
      message: 'Sistema iniciado correctamente',
      time: 'Hace 2 minutos'
    },
    {
      type: 'product',
      icon: 'bi bi-box-seam-fill',
      message: 'Base de datos de productos cargada',
      time: 'Hace 5 minutos'
    },
    {
      type: 'settings',
      icon: 'bi bi-gear-fill',
      message: 'Configuración verificada',
      time: 'Hace 8 minutos'
    }
  ];

  private destroy$ = new Subject<void>();

  constructor(
    private authService: AuthService,
    private dataService: DataService,
    private router: Router
  ) {}

  ngOnInit(): void {
    this.loadDashboardData();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private loadDashboardData(): void {
    // Cargar productos
    this.dataService.products$
      .pipe(takeUntil(this.destroy$))
      .subscribe((products: any) => {
        this.totalProducts = products.length;
        this.activeProducts = products.filter((p: { active: boolean; }) => p.active !== false).length;
        this.productsWithDiscount = products.filter((p: { discount: number; }) => p.discount > 0).length;
        
        // Calcular categorías
        const categories = [...new Set(products.map((p: { category: any; }) => p.category))];
        this.totalCategories = categories.length;
        
        // Encontrar categoría más popular
        const categoryCounts = categories.map(cat => ({
          category: cat,
          count: products.filter((p: { category: unknown; }) => p.category === cat).length
        }));
        const mostPopular: any = categoryCounts.sort((a, b) => b.count - a.count)[0];
        this.mostPopularCategory = mostPopular ? this.getCategoryDisplayName(mostPopular.category) : 'N/A';
      });

    // Cargar configuración
    this.dataService.settings$
      .pipe(takeUntil(this.destroy$))
      .subscribe((settings: any) => {
        if (settings) {
          this.whatsappNumber = settings.contact.whatsapp;
        }
      });
  }

  getCurrentTime(): string {
    return new Date().toLocaleString('es-CO', {
      timeZone: 'America/Bogota',
      hour: '2-digit',
      minute: '2-digit',
      day: '2-digit',
      month: '2-digit',
      year: 'numeric'
    });
  }

  private getCategoryDisplayName(category: string): string {
    const names: Record<string, string> = {
      'desechable': 'Desechables',
      'pod': 'Pod Systems',
      'liquid': 'E-Liquids',
      'mod': 'Mods',
      'kit': 'Kits',
      'recargable': 'Recargables'
    };
    return names[category] || category;
  }

  logout(): void {
    this.authService.logout();
    this.router.navigate(['/admin/login']);
  }

  exportData(): void {
    const products = this.dataService.getProducts();
    const settings = this.dataService.getSettings();
    
    const exportData = {
      products,
      settings,
      exportedAt: new Date().toISOString()
    };

    const dataStr = JSON.stringify(exportData, null, 2);
    const dataBlob = new Blob([dataStr], { type: 'application/json' });
    
    const url = URL.createObjectURL(dataBlob);
    const link = document.createElement('a');
    link.href = url;
    link.download = `vaplure-backup-${new Date().toISOString().split('T')[0]}.json`;
    link.click();
    
    URL.revokeObjectURL(url);
  }

  viewSite(): void {
    window.open('/inicio', '_blank');
  }
}

export class AdminLoginComponent implements OnInit {
  password = '';
  hasError = false;
  errorMessage = '';
  isLoading = false;

  constructor(
    private authService: AuthService,
    private router: Router
  ) {}

  ngOnInit(): void {
    // Si ya está autenticado, redirigir al dashboard
    if (this.authService.isAuthenticated()) {
      this.router.navigate(['/admin/dashboard']);
    }
  }

  login(): void {
    if (!this.password) {
      this.showError('Por favor ingresa tu contraseña');
      return;
    }

    this.isLoading = true;
    this.hasError = false;

    // Simular delay de autenticación
    setTimeout(() => {
      const success = this.authService.login(this.password);
      
      if (success) {
        this.router.navigate(['/admin/dashboard']);
      } else {
        this.showError('Contraseña incorrecta. Intenta nuevamente.');
        this.password = '';
      }
      
      this.isLoading = false;
    }, 800);
  }

  private showError(message: string): void {
    this.hasError = true;
    this.errorMessage = message;
    
    // Limpiar error después de 5 segundos
    setTimeout(() => {
      this.hasError = false;
      this.errorMessage = '';
    }, 5000);
  }
}

// src/app/modules/admin/products/product-management.component.ts
import { Component, OnInit, OnDestroy } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule, ReactiveFormsModule, FormBuilder, FormGroup, Validators, FormArray } from '@angular/forms';
import { RouterModule } from '@angular/router';
import { Subject, takeUntil } from 'rxjs';
import { DataService } from '../../../../service/data.service';
import { AuthService } from '../../../../service/auth.service';
import { CreateProductRequest, Product } from '../../../../models/product.model';

@Component({
    selector: 'app-product-management',
    standalone: true,
    imports: [CommonModule, FormsModule, ReactiveFormsModule, RouterModule],
    templateUrl: './product-management.component.html',
    styleUrls: ['./product-management.component.scss']
})
export class ProductManagementComponent implements OnInit, OnDestroy {
    products: Product[] = [];
    filteredProducts: Product[] = [];

    // Form
    productForm!: FormGroup;
    showModal = false;
    editingProduct: Product | null = null;

    // Delete confirmation
    showDeleteModal = false;
    productToDelete: Product | null = null;

    // Filters
    searchTerm = '';
    selectedCategory = '';
    selectedStatus = '';

    // Stats
    activeProducts = 0;
    productsWithDiscount = 0;

    private destroy$ = new Subject<void>();

    constructor(
        private dataService: DataService,
        private authService: AuthService,
        private fb: FormBuilder
    ) {
        this.initForm();
    }

    ngOnInit(): void {
        this.loadProducts();
    }

    ngOnDestroy(): void {
        this.destroy$.next();
        this.destroy$.complete();
    }

    private initForm(): void {
        this.productForm = this.fb.group({
            name: ['', [Validators.required]],
            description: ['', [Validators.required]],
            category: ['', [Validators.required]],
            image: [''],
            imageAlt: [''],
            price: [0, [Validators.required, Validators.min(0)]],
            discount: [0, [Validators.min(0), Validators.max(100)]],
            stock: [0, [Validators.min(0)]],
            active: [true],
            puffs: [''],
            nicotine: [''],
            battery: [''],
            flavors: this.fb.array([]),
            features: this.fb.array([])
        });
    }

    get flavorsArray(): FormArray {
        return this.productForm.get('flavors') as FormArray;
    }

    get featuresArray(): FormArray {
        return this.productForm.get('features') as FormArray;
    }

    private loadProducts(): void {
        this.dataService.products$
            .pipe(takeUntil(this.destroy$))
            .subscribe(products => {
                this.products = products;
                this.updateStats();
                this.filterProducts();
            });
    }

    private updateStats(): void {
        this.activeProducts = this.products.filter(p => (p as any).active !== false).length;
        this.productsWithDiscount = this.products.filter(p => p.discount > 0).length;
    }

    filterProducts(): void {
        let filtered = [...this.products];

        // Search filter
        if (this.searchTerm) {
            const term = this.searchTerm.toLowerCase();
            filtered = filtered.filter(p =>
                p.name.toLowerCase().includes(term) ||
                p.description.toLowerCase().includes(term) ||
                p.category.toLowerCase().includes(term)
            );
        }

        // Category filter
        if (this.selectedCategory) {
            filtered = filtered.filter(p => p.category === this.selectedCategory);
        }

        // Status filter
        if (this.selectedStatus) {
            if (this.selectedStatus === 'active') {
                filtered = filtered.filter(p => (p as any).active !== false);
            } else if (this.selectedStatus === 'inactive') {
                filtered = filtered.filter(p => (p as any).active === false);
            }
        }

        this.filteredProducts = filtered;
    }

    showProductForm(product?: Product): void {
        this.editingProduct = product || null;
        this.resetForm();

        if (product) {
            this.populateForm(product);
        } else {
            // Add initial flavor and feature
            this.addFlavor();
            this.addFeature();
        }

        this.showModal = true;
    }

    private resetForm(): void {
        this.productForm.reset({
            name: '',
            description: '',
            category: '',
            image: '',
            imageAlt: '',
            price: 0,
            discount: 0,
            stock: 0,
            active: true,
            puffs: '',
            nicotine: '',
            battery: ''
        });

        // Clear arrays
        while (this.flavorsArray.length) {
            this.flavorsArray.removeAt(0);
        }
        while (this.featuresArray.length) {
            this.featuresArray.removeAt(0);
        }
    }

    private populateForm(product: Product): void {
        this.productForm.patchValue({
            name: product.name,
            description: product.description,
            category: product.category,
            image: product.image,
            imageAlt: product.imageAlt || '',
            price: product.price,
            discount: product.discount,
            stock: (product as any).stock || 0,
            active: (product as any).active !== false,
            puffs: product.puffs,
            nicotine: product.nicotine,
            battery: product.battery
        });

        // Populate flavors
        product.flavors.forEach(flavor => {
            this.flavorsArray.push(this.fb.control(flavor));
        });

        // Populate features
        product.features.forEach(feature => {
            this.featuresArray.push(this.fb.control(feature));
        });
    }

    addFlavor(): void {
        this.flavorsArray.push(this.fb.control(''));
    }

    removeFlavor(index: number): void {
        this.flavorsArray.removeAt(index);
    }

    addFeature(): void {
        this.featuresArray.push(this.fb.control(''));
    }

    removeFeature(index: number): void {
        this.featuresArray.removeAt(index);
    }

    saveProduct(): void {
        if (this.productForm.valid) {
            const formValue = this.productForm.value;

            // Filter empty flavors and features
            const flavors = formValue.flavors.filter((f: string) => f.trim());
            const features = formValue.features.filter((f: string) => f.trim());

            // Usar el tipo CreateProductRequest para nuevos productos
            const productData: CreateProductRequest = {
                name: formValue.name,
                description: formValue.description,
                category: formValue.category,
                image: formValue.image || '',
                imageAlt: formValue.imageAlt || formValue.name,
                price: formValue.price,
                discount: formValue.discount || 0,
                stock: formValue.stock || 0,
                active: formValue.active !== false,
                puffs: formValue.puffs || '',
                nicotine: formValue.nicotine || '',
                battery: formValue.battery || '',
                flavors,
                features
            };

            if (this.editingProduct) {
                // Para actualizaciones, usar Partial<Product>
                const updateData: Partial<Product> = {
                    ...productData,
                    // No incluir createdAt en actualizaciones
                    updatedAt: new Date().toISOString()
                };
                this.dataService.updateProduct(this.editingProduct.id, updateData);
            } else {
                // Para creación, usar CreateProductRequest
                this.dataService.addProduct(productData);
            }

            this.closeModal();
        }
    }

    closeModal(): void {
        this.showModal = false;
        this.editingProduct = null;
    }

    editProduct(product: Product): void {
        this.showProductForm(product);
    }

    toggleProductStatus(product: Product): void {
        const newStatus = !(product as any).active;
        this.dataService.updateProduct(product.id, { active: newStatus } as any);
    }

    confirmDelete(product: Product): void {
        this.productToDelete = product;
        this.showDeleteModal = true;
    }

    deleteProduct(): void {
        if (this.productToDelete) {
            this.dataService.deleteProduct(this.productToDelete.id);
            this.cancelDelete();
        }
    }

    cancelDelete(): void {
        this.showDeleteModal = false;
        this.productToDelete = null;
    }

    getFinalPrice(product: Product): number {
        return product.price * (1 - product.discount / 100);
    }

    getCategoryDisplayName(category: string): string {
        const names: Record<string, string> = {
            'desechable': 'Desechables',
            'pod': 'Pod Systems',
            'liquid': 'E-Liquids',
            'mod': 'Mods',
            'kit': 'Kits',
            'recargable': 'Recargables'
        };
        return names[category] || category;
    }

    onImageError(event: any): void {
        event.target.style.display = 'none';
        const fallback = event.target.nextElementSibling;
        if (fallback) {
            fallback.style.display = 'flex';
        }
    }

    trackByProductId(index: number, product: Product): number {
        return product.id;
    }

    private getProductProperty<K extends keyof Product>(product: Product, key: K, defaultValue: any): any {
        return product[key] !== undefined ? product[key] : defaultValue;
    }
}

export class SettingsManagementComponent implements OnInit, OnDestroy {
  contactForm!: FormGroup;
  deliveryForm!: FormGroup;
  scheduleForm!: FormGroup;
  siteForm!: FormGroup;
  
  hasChanges = false;
  originalSettings: Settings | null = null;
  
  weekDays = [
    { key: 'monday', label: 'Lunes' },
    { key: 'tuesday', label: 'Martes' },
    { key: 'wednesday', label: 'Miércoles' },
    { key: 'thursday', label: 'Jueves' },
    { key: 'friday', label: 'Viernes' },
    { key: 'saturday', label: 'Sábado' },
    { key: 'sunday', label: 'Domingo' }
  ];
  
  private destroy$ = new Subject<void>();

  constructor(
    private dataService: DataService,
    private authService: AuthService,
    private fb: FormBuilder
  ) {
    this.initForms();
  }

  ngOnInit(): void {
    this.loadSettings();
    this.watchForChanges();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private initForms(): void {
    // Contact Form
    this.contactForm = this.fb.group({
      whatsapp: ['', [Validators.required, Validators.pattern(/^\d{10}$/)]],
      email: ['', [Validators.email]]
    });

    // Delivery Form
    this.deliveryForm = this.fb.group({
      city: [''],
      deliveryTime: [''],
      deliveryFee: [0, [Validators.min(0)]],
      freeDeliveryMinimum: [0, [Validators.min(0)]],
      availableNightDelivery: [false]
    });

    // Schedule Form
    const scheduleControls: any = {};
    this.weekDays.forEach(day => {
      scheduleControls[day.key + '_isOpen'] = [true];
      scheduleControls[day.key + '_open'] = ['09:00'];
      scheduleControls[day.key + '_close'] = ['22:00'];
    });
    this.scheduleForm = this.fb.group(scheduleControls);

    // Site Form
    this.siteForm = this.fb.group({
      name: [''],
      slogan: [''],
      description: [''],
      logo: [''],
      primaryColor: ['#00ff88'],
      secondaryColor: ['#4c6ef5'],
      accentColor: ['#00d4ff']
    });
  }

  private loadSettings(): void {
    this.dataService.settings$
      .pipe(takeUntil(this.destroy$))
      .subscribe((settings: any) => {
        if (settings) {
          this.originalSettings = JSON.parse(JSON.stringify(settings));
          this.populateForms(settings);
        }
      });
  }

  private populateForms(settings: Settings): void {
    // Populate contact form
    this.contactForm.patchValue({
      whatsapp: settings.contact.whatsapp,
      email: settings.contact.email
    });

    // Populate delivery form
    this.deliveryForm.patchValue({
      city: settings.contact.deliveryInfo.city,
      deliveryTime: settings.contact.deliveryInfo.deliveryTime,
      deliveryFee: settings.contact.deliveryInfo.deliveryFee,
      freeDeliveryMinimum: settings.contact.deliveryInfo.freeDeliveryMinimum,
      availableNightDelivery: settings.contact.deliveryInfo.availableNightDelivery
    });

    // Populate schedule form
    Object.keys(settings.contact.schedule).forEach(day => {
      const schedule = settings.contact.schedule[day];
      this.scheduleForm.patchValue({
        [day + '_isOpen']: schedule.isOpen,
        [day + '_open']: schedule.open,
        [day + '_close']: schedule.close
      });
    });

    // Populate site form
    this.siteForm.patchValue({
      name: settings.site.name,
      slogan: settings.site.slogan,
      description: settings.site.description,
      logo: settings.site.logo,
      primaryColor: settings.site.primaryColor,
      secondaryColor: settings.site.secondaryColor,
      accentColor: settings.site.accentColor
    });
  }

  private watchForChanges(): void {
    // Watch all forms for changes
    [this.contactForm, this.deliveryForm, this.scheduleForm, this.siteForm].forEach(form => {
      form.valueChanges
        .pipe(takeUntil(this.destroy$))
        .subscribe(() => {
          this.checkForChanges();
        });
    });
  }

  private checkForChanges(): void {
    if (!this.originalSettings) return;

    const currentSettings = this.buildSettingsFromForms();
    this.hasChanges = JSON.stringify(currentSettings) !== JSON.stringify(this.originalSettings);
  }

  private buildSettingsFromForms(): Settings {
    const contactValue = this.contactForm.value;
    const deliveryValue = this.deliveryForm.value;
    const scheduleValue = this.scheduleForm.value;
    const siteValue = this.siteForm.value;

    // Build schedule object
    const schedule: any = {};
    this.weekDays.forEach(day => {
      schedule[day.key] = {
        isOpen: scheduleValue[day.key + '_isOpen'],
        open: scheduleValue[day.key + '_open'],
        close: scheduleValue[day.key + '_close']
      };
    });

    return {
      contact: {
        whatsapp: contactValue.whatsapp,
        email: contactValue.email,
        schedule,
        deliveryInfo: {
          city: deliveryValue.city,
          deliveryTime: deliveryValue.deliveryTime,

          freeDeliveryMinimum: deliveryValue.freeDeliveryMinimum,
          availableNightDelivery: deliveryValue.availableNightDelivery
        }

        description: siteValue.description,
        logo: siteValue.logo,
        primaryColor: siteValue.primaryColor,
        secondaryColor: siteValue.secondaryColor,
        accentColor: siteValue.accentColor
      }
    };
  }

  saveAllSettings(): void {
    if (!this.hasChanges || this.hasFormErrors()) return;

    const newSettings = this.buildSettingsFromForms();
    this.dataService.updateSettings(newSettings);
    this.originalSettings = JSON.parse(JSON.stringify(newSettings));
    this.hasChanges = false;

    // Show success message (you can implement a toast service)
    console.log('Configuración guardada exitosamente');
  }

  private hasFormErrors(): boolean {
    return this.contactForm.invalid || 
           this.deliveryForm.invalid || 
           this.scheduleForm.invalid || 
           this.siteForm.invalid;
  }

  discardChanges(): void {
    if (this.originalSettings) {
      this.populateForms(this.originalSettings);

    }
  }

export class CatalogComponent implements OnInit, OnDestroy {
  filteredProducts: Product[] = [];
  currentFilter = 'all';
  isLoading = false;
  isRefreshing = false;
  
  private destroy$ = new Subject<void>();

  constructor(
    private productService: ProductService,
    private cartService: CartService
  ) {}

  ngOnInit(): void {
    this.loadInitialProducts();
    this.subscribeToProductChanges();
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }

  private loadInitialProducts(): void {
    this.isLoading = true;
    
    // Simular carga inicial
    setTimeout(() => {
      this.filteredProducts = this.productService.getAllProducts();
      this.currentFilter = this.productService.getCurrentFilter();
      this.isLoading = false;
    }, 500);
  }

  private subscribeToProductChanges(): void {
    this.productService.filteredProducts$
      .pipe(takeUntil(this.destroy$))
      .subscribe(products => {
        this.filteredProducts = products;
        this.currentFilter = this.productService.getCurrentFilter();
      });
  }

  onFilterChange(category: string): void {
    if (this.currentFilter === category) return;
    
    this.isLoading = true;
    this.currentFilter = category;
    
    setTimeout(() => {
      this.productService.filterProducts(category);
      this.isLoading = false;
    }, 300);
  }

  onAddToCart(event: { product: Product, quantity: number }): void {
    this.cartService.addToCart(event.product, event.quantity);
    
    // Mostrar notificación temporal (opcional)
    this.showAddToCartNotification(event.product.name);
  }

  refreshCatalog(): void {
    this.isRefreshing = true;
    
    this.productService.refreshProducts()
      .pipe(
        finalize(() => this.isRefreshing = false),
        takeUntil(this.destroy$)
      )
      .subscribe();
  }

  resetFilter(): void {
    this.onFilterChange('all');
  }

  trackByProductId(index: number, product: Product): number {
    return product.id;
  }

  getCategoryName(category: string): string {
    const categoryNames: Record<string, string> = {
      'all': 'Todos',
      'desechable': 'Desechables',
      'pod': 'Pod Systems',
      'liquid': 'E-Liquids',
      'mod': 'Mods',
      'kit': 'Kits',
      'recargable': 'Recargables'
    };
    return categoryNames[category] || category;
  }

  hasDiscountedProducts(): boolean {
    return this.filteredProducts.some(product => product.discount > 0);
  }

  private showAddToCartNotification(productName: string): void {
    // Implementar notificación temporal (opcional)
    console.log(`${productName} agregado al carrito`);
  }
}

      'tuesday': 'Martes',
      'wednesday': 'Miércoles',
      'thursday': 'Jueves',
      'friday': 'Viernes',

import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';
import * as CryptoJS from 'crypto-js';

@Injectable({
  providedIn: 'root'
})
export class AuthService {
  private isAuthenticatedSubject = new BehaviorSubject<boolean>(false);
  public isAuthenticated$ = this.isAuthenticatedSubject.asObservable();
  
  private readonly SECRET_KEY = 'vaplure-admin-2025';
  private readonly ADMIN_PASSWORD_HASH = '09d0cc6f394e30cc1a446fbcd47639fd431e414681f4f10d1d979cc269a106c9'; // SHA256 de "admin123"

  constructor() {
    this.checkAuthStatus();
  }

  private checkAuthStatus(): void {
    const token = sessionStorage.getItem('admin_token');
    const expiry = sessionStorage.getItem('admin_expiry');
    
    if (token && expiry) {
      const now = new Date().getTime();
      if (now < parseInt(expiry)) {
        this.isAuthenticatedSubject.next(true);
      } else {
        this.logout();
      }
    }
  }

  login(password: string): boolean {
    const passwordHash = CryptoJS.SHA256(password).toString();
    
    if (passwordHash === this.ADMIN_PASSWORD_HASH) {
      const token = this.generateToken();
      const expiry = new Date().getTime() + (30 * 60 * 1000); // 30 minutos
      
      sessionStorage.setItem('admin_token', token);
      sessionStorage.setItem('admin_expiry', expiry.toString());
      
      this.isAuthenticatedSubject.next(true);
      return true;
    }
    
    return false;
  }

  logout(): void {
    sessionStorage.removeItem('admin_token');
    sessionStorage.removeItem('admin_expiry');
    this.isAuthenticatedSubject.next(false);
  }

  isAuthenticated(): boolean {
    return this.isAuthenticatedSubject.value;
  }

  private generateToken(): string {
    const timestamp = new Date().getTime();
    return CryptoJS.AES.encrypt(timestamp.toString(), this.SECRET_KEY).toString();
  }
}

@Injectable({
  providedIn: 'root'

export class CartService {
  private cartItems: CartItem[] = [];
  private cartSubject = new BehaviorSubject<CartItem[]>([]);
  private cartTotalSubject = new BehaviorSubject<number>(0);
  private cartItemCountSubject = new BehaviorSubject<number>(0);

  public cart$ = this.cartSubject.asObservable();
  public cartTotal$ = this.cartTotalSubject.asObservable();
  public cartItemCount$ = this.cartItemCountSubject.asObservable();

  constructor(private dataService: DataService) {}

  addToCart(product: Product, quantity: number = 1): void {
    const finalPrice = product.price * (1 - product.discount / 100);
    const existingItem = this.cartItems.find(item => item.id === product.id);

    if (existingItem) {
      existingItem.quantity += quantity;
    } else {
      const cartItem: CartItem = {
        ...product,
        quantity,
        finalPrice
      };
      this.cartItems.push(cartItem);
    }

    this.updateCartObservables();
  }

  removeFromCart(productId: number): void {
    this.cartItems = this.cartItems.filter(item => item.id !== productId);
    this.updateCartObservables();
  }

  updateQuantity(productId: number, quantity: number): void {
    const item = this.cartItems.find(item => item.id === productId);
    if (item && quantity > 0) {
      item.quantity = quantity;
      this.updateCartObservables();
    }
  }

  clearCart(): void {
    this.cartItems = [];
    this.updateCartObservables();
  }

  getCartItems(): CartItem[] {
    return this.cartItems;

      message += `   💰 Precio: ${itemTotal.toLocaleString()}\n\n`;
    });

    message += '━━━━━━━━━━━━━━━━━━━━\n';
    message += `💵 *TOTAL: ${total.toLocaleString()}*\n\n`;
    message += `📍 *Dirección de entrega en ${cityName}:*\n`;
    message += '(Por favor proporciona tu dirección completa)\n\n';
    message += '🕐 *Horario preferido:*\n';
    message += this.getAvailableSchedule() + '\n\n';
    message += '💳 *Método de pago:* Contraentrega\n';
    message += `⚡ *Tiempo estimado:* ${deliveryTime}\n\n`;
    message += `¡Gracias por elegir ${settings?.site.name || 'VAPLURE'}! 🚀`;

    return message;
  }

  private getAvailableSchedule(): string {
    const settings = this.dataService.getSettings();
    if (!settings) return '(Disponible de 9:00 AM a 10:00 PM)';

    const schedule = settings.contact.schedule;
    const today = new Date().toLocaleDateString('en-US', { weekday: 'short' }).toLowerCase(); // mon, tue, etc.
    
    // Mapear días en español
    const dayMap: Record<string, string> = {
      'mon': 'monday',
      'tue': 'tuesday', 
      'wed': 'wednesday',
      'thu': 'thursday',
      'fri': 'friday',
      'sat': 'saturday',
      'sun': 'sunday'
    };

    const todayKey = dayMap[today] || 'monday';
    const todaySchedule = schedule[todayKey];

    if (todaySchedule && todaySchedule.isOpen) {
      return `(Disponible hoy de ${todaySchedule.open} a ${todaySchedule.close})`;
    } else {
      // Encontrar el próximo día disponible
      const dayOrder = ['monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday', 'sunday'];
      const currentIndex = dayOrder.indexOf(todayKey);
      
      for (let i = 1; i <= 7; i++) {
        const nextDayIndex = (currentIndex + i) % 7;
        const nextDay = dayOrder[nextDayIndex];
        const nextSchedule = schedule[nextDay];
        
        if (nextSchedule && nextSchedule.isOpen) {
          const dayNames: Record<string, string> = {
            'monday': 'lunes',
            'tuesday': 'martes',
            'wednesday': 'miércoles',
            'thursday': 'jueves',
            'friday': 'viernes',
            'saturday': 'sábado',
            'sunday': 'domingo'
          };
          
          return `(Próxima disponibilidad: ${dayNames[nextDay]} de ${nextSchedule.open} a ${nextSchedule.close})`;
        }
      }
    }

    return '(Contacta para confirmar horarios de disponibilidad)';
  }
}

  private loadInitialData(): void {
    // Primero intentar cargar desde localStorage
    const storedProducts = this.getFromStorage('products');
    const storedSettings = this.getFromStorage('settings');

    if (storedProducts) {
      this.productsSubject.next(storedProducts);
    } else {
      // Si no hay datos guardados, cargar desde JSON
      this.http.get<{products: Product[]}>('assets/data/products.json').subscribe({
        next: (data) => this.productsSubject.next(data.products),
        error: (error) => {
          console.error('Error loading products:', error);
          // Productos por defecto en caso de error
          this.productsSubject.next(this.getDefaultProducts());
        }
      });
    }

    if (storedSettings) {
      this.settingsSubject.next(storedSettings);
    } else {
      // Si no hay configuración guardada, cargar desde JSON
      this.http.get<Settings>('assets/data/settings.json').subscribe({
        next: (data) => this.settingsSubject.next(data),
        error: (error) => {
          console.error('Error loading settings:', error);
          // Configuración por defecto en caso de error
          this.settingsSubject.next(this.getDefaultSettings());
        }
      });
    }
  }

  getProducts(): Product[] {
    return this.productsSubject.value;
  }

  addProduct(productData: CreateProductRequest): void {
    const products = this.getProducts();
    const newProduct: Product = {
      ...productData,
      id: this.generateId(),
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
      // Valores por defecto si no se proporcionan
      stock: productData.stock ?? 0,
      active: productData.active ?? true,
      imageAlt: productData.imageAlt || productData.name
    };
    
    const updatedProducts = [...products, newProduct];
    this.productsSubject.next(updatedProducts);
    this.saveToStorage('products', updatedProducts);
  }

  updateProduct(id: number, updates: Partial<Product>): void {
    const products = this.getProducts();
    const index = products.findIndex(p => p.id === id);
    
    if (index !== -1) {
      products[index] = { 
        ...products[index], 
        ...updates, 
        updatedAt: new Date().toISOString() 
      };
      this.productsSubject.next([...products]);
      this.saveToStorage('products', products);
    }
  }

  deleteProduct(id: number): void {
    const products = this.getProducts().filter(p => p.id !== id);
    this.productsSubject.next(products);
    this.saveToStorage('products', products);
  }

  // CONFIGURACIÓN

    return this.settingsSubject.value;
  }

    this.settingsSubject.next(settings);
    this.saveToStorage('settings', settings);
  }

  updateContactInfo(contactInfo: Settings['contact']): void {
    const currentSettings = this.getSettings();
    if (currentSettings) {

        ...currentSettings,
        contact: contactInfo
      };

    }

      return stored ? JSON.parse(stored) : null;
    } catch (error) {

    }
  }

import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';
import { DataService } from './data.service';
import { Product } from '../models/product.model';

@Injectable({
  providedIn: 'root'
})
export class ProductService {
  private filteredProductsSubject = new BehaviorSubject<Product[]>([]);
  public filteredProducts$ = this.filteredProductsSubject.asObservable();
  private currentFilter = 'all';

  constructor(private dataService: DataService) {
    // Suscribirse a los productos del DataService
    this.dataService.products$.subscribe(products => {
      this.filterProducts(this.currentFilter);
    });
  }

  getAllProducts(): Product[] {
    return this.dataService.getProducts();
  }

  getProductsByCategory(category: string): Product[] {
    const products = this.getAllProducts();
    if (category === 'all') {
      return products;
    }
    return products.filter(product => product.category === category);
  }

  filterProducts(category: string): void {
    this.currentFilter = category;
    const filtered = this.getProductsByCategory(category);
    this.filteredProductsSubject.next(filtered);
  }

  getCurrentFilter(): string {
    return this.currentFilter;
  }

  getProductById(id: number): Product | undefined {
    return this.getAllProducts().find(product => product.id === id);
  }

  searchProducts(searchTerm: string): Product[] {
    const products = this.getAllProducts();
    if (!searchTerm.trim()) {
      return products;
    }

    const term = searchTerm.toLowerCase();
    return products.filter(product => 
      product.name.toLowerCase().includes(term) ||
      product.description.toLowerCase().includes(term) ||
      product.flavors.some(flavor => flavor.toLowerCase().includes(term)) ||
      product.features.some(feature => feature.toLowerCase().includes(term))
    );
  }

  getDiscountedProducts(): Product[] {
    return this.getAllProducts().filter(product => product.discount > 0);
  }

  getProductsByPriceRange(minPrice: number, maxPrice: number): Product[] {
    return this.getAllProducts().filter(product => {
      const finalPrice = product.price * (1 - product.discount / 100);
      return finalPrice >= minPrice && finalPrice <= maxPrice;
    });
  }

  getCategories(): string[] {
    const products = this.getAllProducts();
    const categories = [...new Set(products.map(product => product.category))];
    return ['all', ...categories];
  }

  getProductStats(): {
    total: number;
    byCategory: Record<string, number>;
    averagePrice: number;
    withDiscount: number;
  } {
    const products = this.getAllProducts();
    const byCategory: Record<string, number> = {};
    let totalPrice = 0;
    let withDiscount = 0;

    products.forEach(product => {
      byCategory[product.category] = (byCategory[product.category] || 0) + 1;
      totalPrice += product.price;
      if (product.discount > 0) {
        withDiscount++;
      }
    });

    return {
      total: products.length,
      byCategory,
      averagePrice: totalPrice / products.length,
      withDiscount
    };
  }

  refreshProducts(): Observable<Product[]> {
    return new Promise<Product[]>((resolve) => {
      setTimeout(() => {
        const filtered = this.getProductsByCategory(this.currentFilter);
        this.filteredProductsSubject.next(filtered);
        resolve(filtered);
      }, 500);
    }) as any;
  }
}
